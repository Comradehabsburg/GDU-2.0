CSA_counterfeiting_clear_all = {
	remove_ideas = {
		CSA_counterfeiting_1
		CSA_counterfeiting_2
		CSA_counterfeiting_3
		CSA_counterfeiting_4
		CSA_counterfeiting_5
	}
}

CSA_counterfeiting_level_up = {
	# level 5 not relevant. Cannot go above.
	custom_effect_tooltip = CSA_counterfeiting_up_tt
	if = {
		limit = { has_idea = CSA_counterfeiting_4 }
		swap_ideas = {
			remove_idea = CSA_counterfeiting_4
			add_idea = CSA_counterfeiting_5
		}
	}
	else_if = {
		limit = { has_idea = CSA_counterfeiting_3 }
		swap_ideas = {
			remove_idea = CSA_counterfeiting_3
			add_idea = CSA_counterfeiting_4
		}
	}
	else_if = {
		limit = { has_idea = CSA_counterfeiting_2 }
		swap_ideas = {
			remove_idea = CSA_counterfeiting_2
			add_idea = CSA_counterfeiting_3
		}
	}
	else_if = {
		limit = { has_idea = CSA_counterfeiting_1 }
		swap_ideas = {
			remove_idea = CSA_counterfeiting_1
			add_idea = CSA_counterfeiting_2
		}		
	}
	else_if = {
		limit = { NOT = { has_idea = CSA_counterfeiting_1 } }
		add_ideas = CSA_counterfeiting_1
	}
}

#Important to put LOWEST first
CSA_counterfeiting_level_down = {
custom_effect_tooltip = CSA_counterfeiting_down_tt
	if = {
		limit = { has_idea = CSA_counterfeiting_1 }
		#Makes the tooltip clearer if player looks at the effect early in the game
		effect_tooltip = {
			swap_ideas = {
				remove_idea = CSA_counterfeiting_1
				add_idea = CSA_counterfeiting_none
			}
		}
		hidden_effect = { remove_ideas = CSA_counterfeiting_1 }
	}
	if = {
		limit = { has_idea = CSA_counterfeiting_2 }
		swap_ideas = {
			remove_idea = CSA_counterfeiting_2
			add_idea = CSA_counterfeiting_1
		}
	}
	if = {
		limit = { has_idea = CSA_counterfeiting_3 }
		swap_ideas = {
			remove_idea = CSA_counterfeiting_3
			add_idea = CSA_counterfeiting_2
		}
	}
	if = {
		limit = { has_idea = CSA_counterfeiting_4 }
		swap_ideas = {
			remove_idea = CSA_counterfeiting_4
			add_idea = CSA_counterfeiting_3
		}
	}
	if = {
		limit = { has_idea = CSA_counterfeiting_5 }
		swap_ideas = {
			remove_idea = CSA_counterfeiting_5
			add_idea = CSA_counterfeiting_4
		}
	}
}
